FakeFile.truncate

camino test_util FakeFile

Truncate (or grow) the file to the specified size.

NOTE: This is not a part of the std.stdio.File API. An equivalent function for production code would be something like:

1 import std.stdio : File;
2 
3 void truncate(File file, long size) {
4     import std.file : FileException;
5 
6     version(Posix) {
7         import core.stdc.errno : errno;
8         import core.sys.posix.unistd: ftruncate;
9 
10         auto result = ftruncate(file.fileno(), size)
11             ? 0
12             : errno();
13     }
14 
15     version(Windows) {
16         import core.sys.windows.windows: SetEndOfFile, GetLastError;
17 
18         file.seek(size);
19         auto result = SetEndOfFile(file.windowsHandle())
20             ? 0
21             : GetLastError();
22     }
23 
24     if (result != 0) {
25         throw new FileException(file.name, result);
26     }
27 }
